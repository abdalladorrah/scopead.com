نموذج البرمجة السريع (Quick Blueprint) ⚡

استخدم هذا الهيكل عند إنشاء صفحة جديدة لربط البيانات بالواجهة:

JavaScript const data = projectData.yourPageKey;
document.getElementById('title-ar').textContent = data.title.ar;
data.items.forEach(item => {
  container.innerHTML += `<div class="card">${item.text.ar}</div>`;
});
